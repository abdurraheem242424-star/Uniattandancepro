
import React, { useState } from 'react';
import { BookOpen, ShieldCheck, Lock, User, ChevronDown, AlertCircle, Eye, EyeOff, UserCircle } from 'lucide-react';
import { User as UserType } from '../types';

interface LoginProps {
  onLogin: (user: UserType) => void;
}

const Login: React.FC<LoginProps> = ({ onLogin }) => {
  const [loading, setLoading] = useState(false);
  const [facultyName, setFacultyName] = useState('');
  const [password, setPassword] = useState('');
  const [showPassword, setShowPassword] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const handleFakeAuth = (e: React.FormEvent) => {
    e.preventDefault();
    setError(null);

    const trimmedName = facultyName.trim();
    if (!trimmedName) {
      setError('Please enter your full faculty name.');
      return;
    }

    if (!password) {
      setError('Please enter your security token.');
      return;
    }

    setLoading(true);

    // Logic: Password must be [Exact Name Entered] + 301
    const correctPassword = `${trimmedName}301`;

    setTimeout(() => {
      if (password === correctPassword) {
        onLogin({
          email: `${trimmedName.toLowerCase().replace(/[\.\s\(\)]/g, '_')}@university.edu`,
          name: trimmedName,
          photoUrl: `https://ui-avatars.com/api/?name=${encodeURIComponent(trimmedName)}&background=4f46e5&color=fff`
        });
      } else {
        setError('Authentication failed. Ensure the name and token are correct.');
      }
      setLoading(false);
    }, 1200);
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-indigo-700 via-indigo-600 to-blue-500 flex items-center justify-center p-6">
      <div className="max-w-md w-full bg-white rounded-3xl shadow-2xl overflow-hidden">
        <div className="p-8 text-center bg-slate-50 border-b">
          <div className="inline-flex items-center justify-center w-20 h-20 rounded-full bg-indigo-100 text-indigo-600 mb-6 animate-pulse">
            <BookOpen size={40} />
          </div>
          <h1 className="text-3xl font-extrabold text-slate-800 tracking-tight">UniAttend</h1>
          <p className="text-indigo-600 font-bold uppercase text-[10px] tracking-[0.2em] mt-1">BSC (hons) Agriculture 301</p>
          <p className="text-slate-500 mt-2 font-medium">Faculty Authentication Portal</p>
        </div>

        <div className="p-8">
          {error && (
            <div className="mb-6 p-4 bg-rose-50 rounded-xl border border-rose-200 flex items-start gap-3 animate-in fade-in slide-in-from-top-2">
              <AlertCircle className="text-rose-600 shrink-0 mt-0.5" size={20} />
              <p className="text-sm text-rose-800 font-medium">
                {error}
              </p>
            </div>
          )}

          <div className="mb-6 p-4 bg-amber-50 rounded-xl border border-amber-200 flex items-start gap-3">
            <ShieldCheck className="text-amber-600 shrink-0 mt-0.5" size={20} />
            <p className="text-xs text-amber-800">
              Authorized university faculty members only. Your password is your <strong>Full Name</strong> followed by <strong>301</strong>.
            </p>
          </div>

          <form onSubmit={handleFakeAuth} className="space-y-5">
            <div className="space-y-2">
              <label className="text-xs font-bold text-slate-500 uppercase ml-1 flex items-center gap-2">
                <UserCircle size={14} />
                Full Faculty Name
              </label>
              <div className="relative">
                <User className="absolute left-3 top-1/2 -translate-y-1/2 text-slate-400" size={18} />
                <input 
                  type="text"
                  value={facultyName}
                  onChange={(e) => {
                    setFacultyName(e.target.value);
                    setError(null);
                  }}
                  placeholder="e.g. Prof. Dr. Mahmood Iqbal"
                  className="w-full pl-10 pr-4 py-3 bg-white border border-slate-200 rounded-xl focus:ring-2 focus:ring-indigo-500 outline-none font-bold text-slate-700 transition-all shadow-sm"
                  autoComplete="name"
                />
              </div>
            </div>

            <div className="space-y-2">
              <label className="text-xs font-bold text-slate-500 uppercase ml-1">Access Token (Password)</label>
              <div className="relative">
                <Lock className="absolute left-3 top-1/2 -translate-y-1/2 text-slate-400" size={18} />
                <input 
                  type={showPassword ? "text" : "password"}
                  value={password}
                  onChange={(e) => setPassword(e.target.value)}
                  placeholder="Full Name + 301"
                  className="w-full pl-10 pr-10 py-3 bg-slate-50 border border-slate-200 rounded-xl outline-none focus:ring-2 focus:ring-indigo-500 transition-all text-sm"
                />
                <button
                  type="button"
                  onClick={() => setShowPassword(!showPassword)}
                  className="absolute right-3 top-1/2 -translate-y-1/2 text-slate-400 hover:text-indigo-600 transition-colors p-1"
                >
                  {showPassword ? <EyeOff size={18} /> : <Eye size={18} />}
                </button>
              </div>
            </div>

            <button
              type="submit"
              disabled={loading}
              className="w-full bg-indigo-600 hover:bg-indigo-700 text-white font-bold py-4 rounded-xl shadow-lg shadow-indigo-200 transition-all flex items-center justify-center gap-2 mt-4 active:scale-95 disabled:opacity-50 disabled:cursor-not-allowed"
            >
              {loading ? (
                <div className="w-6 h-6 border-3 border-white border-t-transparent rounded-full animate-spin" />
              ) : (
                <>Sign In Securely</>
              )}
            </button>
          </form>

          <div className="mt-10 flex flex-col items-center gap-2">
            <div className="flex items-center gap-2 text-[10px] font-black text-slate-300 uppercase tracking-widest bg-slate-50 px-3 py-1.5 rounded-full border border-slate-100">
              <BookOpen size={12} className="text-indigo-400" />
              UniAttend Secure â€¢ Privacy Mode Active
            </div>
          </div>
        </div>
      </div>
    </div>
  );
};

export default Login;
