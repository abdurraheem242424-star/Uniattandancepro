
import React, { useState } from 'react';
import { BookOpen, Search, Trash2, Edit2, Plus, X, Library, Info, Activity } from 'lucide-react';
import { Subject } from '../types';

interface SubjectManagerProps {
  subjects: Subject[];
  onUpdate: (subjects: Subject[], deletedId?: string) => void;
}

const SubjectManager: React.FC<SubjectManagerProps> = ({ subjects, onUpdate }) => {
  const [searchTerm, setSearchTerm] = useState('');
  const [isAdding, setIsAdding] = useState(false);
  const [editingId, setEditingId] = useState<string | null>(null);
  const [newName, setNewName] = useState('');
  const [newType, setNewType] = useState<'Theory' | 'Practical'>('Theory');

  const filteredSubjects = subjects.filter(s => 
    s.name.toLowerCase().includes(searchTerm.toLowerCase())
  );

  const handleAdd = () => {
    if (!newName.trim()) return;
    const newSubject: Subject = {
      id: crypto.randomUUID(),
      name: newName,
      type: newType,
      isPracticalEnabled: newType === 'Practical'
    };
    onUpdate([...subjects, newSubject]);
    setNewName('');
    setNewType('Theory');
    setIsAdding(false);
  };

  const handleUpdate = (id: string) => {
    if (!newName.trim()) return;
    onUpdate(subjects.map(s => s.id === id ? { ...s, name: newName, type: newType, isPracticalEnabled: newType === 'Practical' } : s));
    setEditingId(null);
    setNewName('');
    setNewType('Theory');
  };

  const handleDelete = (id: string) => {
    const subject = subjects.find(s => s.id === id);
    if (window.confirm(`Are you sure you want to delete "${subject?.name}"? ALL associated attendance data for this subject will be permanently removed.`)) {
      onUpdate(subjects.filter(s => s.id !== id), id);
    }
  };

  return (
    <div className="space-y-6 max-w-4xl mx-auto animate-in fade-in duration-300">
      <header className="flex flex-col md:flex-row md:items-center justify-between gap-4">
        <div>
          <h2 className="text-3xl font-extrabold text-slate-800 flex items-center gap-3">
            <Library className="text-indigo-600" size={32} />
            Subject Management
          </h2>
          <p className="text-slate-500">Configure theory and practical attendance registers.</p>
        </div>
        <button 
          onClick={() => {
            setNewName('');
            setNewType('Theory');
            setIsAdding(true);
          }}
          className="bg-indigo-600 hover:bg-indigo-700 text-white px-6 py-3 rounded-xl shadow-lg shadow-indigo-100 transition-all font-bold flex items-center justify-center gap-2"
        >
          <Plus size={20} />
          Add Subject
        </button>
      </header>

      {/* Search bar */}
      <div className="relative">
        <Search className="absolute left-4 top-1/2 -translate-y-1/2 text-slate-400" size={20} />
        <input 
          type="text" 
          placeholder="Search subjects..."
          value={searchTerm}
          onChange={(e) => setSearchTerm(e.target.value)}
          className="w-full pl-12 pr-4 py-3 bg-white border border-slate-200 rounded-2xl focus:ring-2 focus:ring-indigo-500 outline-none transition-all shadow-sm font-medium text-slate-700"
        />
      </div>

      {/* Grid of Subjects */}
      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        {/* Inline Add/Edit Form Card */}
        {isAdding && (
          <div className="bg-indigo-50 border-2 border-dashed border-indigo-200 p-6 rounded-2xl flex flex-col gap-4 animate-in zoom-in-95 duration-200">
            <h4 className="font-extrabold text-indigo-700 flex items-center gap-2">
              <Plus size={18} /> Add New Register
            </h4>
            <div className="space-y-3">
              <div className="space-y-1">
                <label className="text-[10px] font-bold text-indigo-400 uppercase tracking-widest ml-1">Subject Name</label>
                <input 
                  autoFocus
                  type="text" 
                  placeholder="e.g. Agronomy II"
                  value={newName}
                  onChange={(e) => setNewName(e.target.value)}
                  className="w-full px-4 py-2 rounded-xl border border-indigo-100 focus:ring-2 focus:ring-indigo-500 outline-none bg-white font-medium shadow-sm"
                />
              </div>
              <div className="space-y-1">
                <label className="text-[10px] font-bold text-indigo-400 uppercase tracking-widest ml-1">Category</label>
                <div className="flex bg-white p-1 rounded-xl border border-indigo-100 shadow-sm">
                   <button 
                     onClick={() => setNewType('Theory')}
                     className={`flex-1 py-2 text-xs font-bold rounded-lg transition-all ${newType === 'Theory' ? 'bg-indigo-600 text-white shadow-md' : 'text-slate-400'}`}
                   >
                     Theory
                   </button>
                   <button 
                     onClick={() => setNewType('Practical')}
                     className={`flex-1 py-2 text-xs font-bold rounded-lg transition-all ${newType === 'Practical' ? 'bg-indigo-600 text-white shadow-md' : 'text-slate-400'}`}
                   >
                     Practical
                   </button>
                </div>
              </div>
            </div>
            <div className="flex gap-2 mt-2">
              <button 
                onClick={handleAdd}
                className="flex-1 bg-indigo-600 hover:bg-indigo-700 text-white py-2.5 rounded-xl font-bold shadow-md transition-all active:scale-95"
              >
                Create Register
              </button>
              <button 
                onClick={() => setIsAdding(false)}
                className="px-4 bg-white text-slate-500 border border-slate-200 py-2.5 rounded-xl font-bold hover:bg-slate-50 transition-all"
              >
                Cancel
              </button>
            </div>
          </div>
        )}

        {filteredSubjects.length === 0 && !isAdding ? (
          <div className="col-span-full py-20 text-center bg-white rounded-3xl border border-dashed border-slate-200">
            <div className="w-16 h-16 bg-slate-50 rounded-full flex items-center justify-center mx-auto text-slate-300 mb-4">
              <Library size={32} />
            </div>
            <p className="text-slate-400 font-medium">No subjects matched your search.</p>
          </div>
        ) : (
          filteredSubjects.map(subject => (
            <div key={subject.id} className="bg-white p-6 rounded-3xl border border-slate-100 shadow-sm flex items-center justify-between group hover:border-indigo-200 transition-all">
              {editingId === subject.id ? (
                <div className="flex-1 mr-4 space-y-3">
                  <input 
                    type="text" 
                    value={newName}
                    onChange={(e) => setNewName(e.target.value)}
                    className="w-full px-4 py-2 rounded-xl border border-indigo-300 outline-none text-sm font-bold shadow-inner bg-slate-50"
                  />
                  <div className="flex bg-slate-100 p-1 rounded-xl border border-slate-200">
                    <button 
                      onClick={() => setNewType('Theory')}
                      className={`flex-1 py-1 text-[10px] font-black rounded-lg transition-all uppercase ${newType === 'Theory' ? 'bg-white text-indigo-600 shadow-sm' : 'text-slate-400'}`}
                    >
                      Theory
                    </button>
                    <button 
                      onClick={() => setNewType('Practical')}
                      className={`flex-1 py-1 text-[10px] font-black rounded-lg transition-all uppercase ${newType === 'Practical' ? 'bg-white text-indigo-600 shadow-sm' : 'text-slate-400'}`}
                    >
                      Practical
                    </button>
                  </div>
                  <div className="flex gap-2">
                    <button onClick={() => handleUpdate(subject.id)} className="bg-indigo-600 text-white px-4 py-1.5 rounded-lg text-xs font-bold shadow-sm">Save</button>
                    <button onClick={() => setEditingId(null)} className="bg-slate-200 text-slate-600 px-4 py-1.5 rounded-lg text-xs font-bold">Cancel</button>
                  </div>
                </div>
              ) : (
                <div className="flex items-center gap-4">
                  <div className={`w-12 h-12 rounded-2xl flex items-center justify-center font-bold border ${
                    subject.type === 'Practical' 
                      ? 'bg-emerald-50 text-emerald-600 border-emerald-100' 
                      : 'bg-indigo-50 text-indigo-600 border-indigo-100'
                  }`}>
                    {subject.type === 'Practical' ? <Activity size={24} /> : <BookOpen size={24} />}
                  </div>
                  <div>
                    <h4 className="font-extrabold text-slate-800 group-hover:text-indigo-600 transition-colors">{subject.name}</h4>
                    <span className={`text-[10px] font-black uppercase tracking-widest ${
                      subject.type === 'Practical' ? 'text-emerald-500' : 'text-indigo-400'
                    }`}>
                      {subject.type} REGISTER
                    </span>
                  </div>
                </div>
              )}
              
              {!editingId && (
                <div className="flex items-center gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
                  <button 
                    onClick={() => {
                      setEditingId(subject.id);
                      setNewName(subject.name);
                      setNewType(subject.type);
                    }}
                    className="p-2.5 text-indigo-600 hover:bg-indigo-50 rounded-xl transition-colors"
                  >
                    <Edit2 size={20} />
                  </button>
                  <button 
                    onClick={() => handleDelete(subject.id)}
                    className="p-2.5 text-rose-600 hover:bg-rose-50 rounded-xl transition-colors"
                  >
                    <Trash2 size={20} />
                  </button>
                </div>
              )}
            </div>
          ))
        )}
      </div>

      <div className="p-4 bg-amber-50 rounded-3xl border border-amber-100 flex gap-3 items-start">
        <Info className="text-amber-600 shrink-0 mt-0.5" size={20} />
        <div>
          <h4 className="text-sm font-bold text-amber-800">Critical Data Notice</h4>
          <p className="text-xs text-amber-700 leading-relaxed mt-1">
            Deleting a subject will <strong>purge all historical attendance data</strong> associated with it. This action cannot be undone. If you just want to rename a subject, use the edit button instead.
          </p>
        </div>
      </div>
    </div>
  );
};

export default SubjectManager;
