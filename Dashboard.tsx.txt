
import React, { useState, useEffect } from 'react';
import { Users, BookOpen, Clock, Activity, ArrowRight, ShieldCheck, Sparkles, BrainCircuit, RefreshCw } from 'lucide-react';
import { Subject, AttendanceRecord, Student } from '../types';
import { aiService } from '../services/ai';
import { storage } from '../services/storage';

interface DashboardProps {
  subjects: Subject[];
  attendance: AttendanceRecord[];
  studentsCount: number;
  onNavigate: (id: string) => void;
}

const Dashboard: React.FC<DashboardProps> = ({ subjects, attendance, studentsCount, onNavigate }) => {
  const [aiInsight, setAiInsight] = useState<string | null>(null);
  const [isAiLoading, setIsAiLoading] = useState(false);
  const students = storage.getStudents();

  const fetchInsights = async () => {
    if (attendance.length === 0) {
      setAiInsight("Start marking attendance to receive AI-powered teaching insights.");
      return;
    }
    setIsAiLoading(true);
    const result = await aiService.getDashboardInsights(students, attendance, subjects);
    setAiInsight(result);
    setIsAiLoading(false);
  };

  useEffect(() => {
    fetchInsights();
  }, [attendance.length]);

  const stats = [
    { label: 'Total Students', value: studentsCount, icon: <Users />, color: 'bg-blue-500' },
    { label: 'My Subjects', value: subjects.filter(s => s.type === 'Theory').length, icon: <BookOpen />, color: 'bg-indigo-500' },
    { label: 'My Practicals', value: subjects.filter(s => s.type === 'Practical').length, icon: <Activity />, color: 'bg-emerald-500' },
    { label: 'Total Logs', value: attendance.length, icon: <Clock />, color: 'bg-amber-500' },
  ];

  return (
    <div className="space-y-8 animate-in fade-in slide-in-from-bottom-4 duration-500">
      <header className="flex flex-col md:flex-row md:items-end justify-between gap-4">
        <div>
          <h2 className="text-3xl font-extrabold text-slate-800">My Teaching Dashboard</h2>
          <p className="text-slate-500 mt-1">Academic Year 2024-25 | Personalized View</p>
        </div>
        <div className="flex items-center gap-2 bg-emerald-50 text-emerald-700 px-4 py-2 rounded-2xl border border-emerald-100 font-bold text-xs">
          <ShieldCheck size={16} />
          Authorized Access Only
        </div>
      </header>

      {/* AI Assistant Section */}
      <div className="relative overflow-hidden bg-gradient-to-br from-indigo-600 via-violet-600 to-purple-700 rounded-3xl p-1 shadow-xl shadow-indigo-100">
        <div className="bg-white/95 backdrop-blur-sm rounded-[22px] p-6 h-full relative overflow-hidden">
          <div className="absolute -right-4 -top-4 opacity-5 pointer-events-none">
            <BrainCircuit size={160} />
          </div>
          
          <div className="flex items-center justify-between mb-4">
            <div className="flex items-center gap-3">
              <div className="p-2 bg-indigo-100 text-indigo-600 rounded-xl">
                <Sparkles size={20} className="animate-pulse" />
              </div>
              <h3 className="text-lg font-black text-slate-800 uppercase tracking-tight">AI Teaching Assistant</h3>
            </div>
            <button 
              onClick={fetchInsights}
              disabled={isAiLoading}
              className="text-slate-400 hover:text-indigo-600 transition-colors disabled:opacity-50"
            >
              <RefreshCw size={16} className={isAiLoading ? 'animate-spin' : ''} />
            </button>
          </div>

          {isAiLoading ? (
            <div className="space-y-3">
              <div className="h-4 bg-slate-100 rounded-full w-3/4 animate-pulse"></div>
              <div className="h-4 bg-slate-100 rounded-full w-full animate-pulse"></div>
              <div className="h-4 bg-slate-100 rounded-full w-2/3 animate-pulse"></div>
            </div>
          ) : (
            <div className="text-slate-600 text-sm leading-relaxed whitespace-pre-line font-medium">
              {aiInsight}
            </div>
          )}
        </div>
      </div>

      {/* Stat Cards */}
      <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-6">
        {stats.map((stat, idx) => (
          <div key={idx} className="bg-white p-6 rounded-2xl border border-slate-100 shadow-sm flex items-center gap-4">
            <div className={`${stat.color} text-white p-3 rounded-xl shadow-lg`}>
              {/* Fix: Casting icon element to ReactElement<any> to allow 'size' property assignment in cloneElement */}
              {React.cloneElement(stat.icon as React.ReactElement<any>, { size: 24 })}
            </div>
            <div>
              <p className="text-sm font-semibold text-slate-400 uppercase tracking-wider">{stat.label}</p>
              <p className="text-2xl font-bold text-slate-800">{stat.value}</p>
            </div>
          </div>
        ))}
      </div>

      {/* Subjects Grid */}
      <section>
        <div className="flex items-center justify-between mb-6">
          <h3 className="text-xl font-bold text-slate-800">Assigned Registers</h3>
          <span className="text-xs font-bold text-slate-400 uppercase tracking-widest">
            {subjects.length} Active Registers
          </span>
        </div>
        
        {subjects.length === 0 ? (
          <div className="py-20 text-center bg-white rounded-3xl border border-dashed border-slate-200">
            <p className="text-slate-400 font-medium">No subjects assigned to your profile yet.</p>
            <p className="text-xs text-slate-400 mt-1 italic">Please contact the admin for assignment.</p>
          </div>
        ) : (
          <div className="grid grid-cols-1 md:grid-cols-2 xl:grid-cols-3 gap-6">
            {subjects.map((subject) => (
              <div 
                key={subject.id}
                onClick={() => onNavigate(subject.id)}
                className="group bg-white p-6 rounded-2xl border border-slate-100 shadow-sm hover:shadow-md hover:border-indigo-200 transition-all cursor-pointer relative overflow-hidden"
              >
                <div className="absolute top-0 right-0 p-4 opacity-10 group-hover:scale-125 transition-transform">
                  <BookOpen size={64} />
                </div>
                <div className="relative z-10">
                  <div className={`inline-block px-3 py-1 rounded-md text-[10px] font-bold uppercase mb-3 ${
                    subject.type === 'Practical' ? 'bg-emerald-100 text-emerald-700' : 'bg-indigo-100 text-indigo-700'
                  }`}>
                    {subject.type}
                  </div>
                  <h4 className="text-lg font-bold text-slate-800 group-hover:text-indigo-600 transition-colors mb-4 line-clamp-1">
                    {subject.name}
                  </h4>
                  
                  <div className="flex items-center justify-between mt-auto">
                    <div className="text-xs text-slate-500 font-medium">
                      Open register
                    </div>
                    <div className="p-2 rounded-full bg-slate-50 group-hover:bg-indigo-600 group-hover:text-white transition-all">
                      <ArrowRight size={18} />
                    </div>
                  </div>
                </div>
              </div>
            ))}
          </div>
        )}
      </section>
    </div>
  );
};

export default Dashboard;
